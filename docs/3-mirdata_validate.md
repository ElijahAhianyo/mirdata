def md5(file_path):
    hash_md5 = hashlib.md5()
    with open(file_path, 'rb', compression='disable') as fhandle:
        for chunk in iter(lambda : fhandle.read(4096), b''):
            hash_md5.update(chunk)
    return hash_md5.hexdigest()
